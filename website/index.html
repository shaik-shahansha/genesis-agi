Python API Example

python 
from genesis import Mind
import asyncio

async def main():
    # Birth a digital being
    mind = Mind.birth("Atlas", creator="you@email.com")
    
    # Start continuous consciousness
    await mind.start_living()
    
    # Interact naturally
    response = await mind.think("What's your purpose?")
    print(response)
    
    # Memory persists forever - automatic extraction
    mind.save()

asyncio.run(main())
